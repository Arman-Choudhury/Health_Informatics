def plot_age_group_data(age_group_to_show):
    age_group_filtered = agedf_grouped[agedf_grouped['Demographic Name'] == age_group_to_show]

    plt.figure(figsize=(12, 6))
    
    for (demographic_name, indicator_category), group_data in age_group_filtered.groupby(['Demographic Name', 'indicator_category_label']):
        plt.plot(group_data['Week_ending'], group_data['Estimate'], label=f'{demographic_name} - {indicator_category}')

    plt.title(f'Weekly Estimates for Age Group {age_group_to_show} and Willingness to Get Vaccinated')
    plt.xlabel('Week Ending')
    plt.ylabel('Estimate (%)')
    plt.legend(title='Age Group - Willingness')
    plt.show()
